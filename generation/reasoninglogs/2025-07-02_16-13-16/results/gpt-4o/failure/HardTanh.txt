transformer deeppoly{
    HardTanh ->
      ((prev[u]) <= -1) ? (-1, -1, -1, -1) :
      ((prev[l]) >= 1) ? (1, 1, 1, 1) :
      (((prev[l]) >= -1) & ((prev[u]) <= 1)) ? (prev[l], prev[u], prev[L], prev[U]) :
      (((prev[l]) < -1) & ((prev[u]) <= 1)) ? (-1, prev[u], -1, prev[U]) :
      (((prev[l]) >= -1) & ((prev[u]) > 1)) ? (prev[l], 1, prev[L], 1) :
      ((prev[l]) < -1 & (prev[u]) > 1) ? (-1, 1, -1, 1) :
      (prev[l], prev[u], prev[L], prev[U]);
}
