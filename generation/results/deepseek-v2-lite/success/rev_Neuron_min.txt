transformer deeppoly{
    RevMin -> (curr[l]<=0 && curr[u]<=0) ? (curr[u], curr[l], curr, curr) : 
              (curr[l]>=0 && curr[u]>=0) ? (curr[l], curr[u], curr, curr) : 
              (curr[l]<=0 && curr[u]>=0) ? (0, max(prev[u], -prev[l]), prev, prev*(prev[u]+prev[l])/(prev[u]-prev[l]) - (((2*prev[u])*prev[l])/(prev[u]-prev[l]))) : 
              (curr[l]>=0 && curr[u]<=0) ? (0, min(prev[u], -prev[l]), prev, prev*(prev[u]+prev[l])/(prev[u]-prev[l]) - (((2*prev[u])*prev[l])/(prev[u]-prev[l]))) : (0, 0, 0, 0);
}
