transformer deeppoly{
    HardSwish ->
        ((prev[l] >= 3) ? 
            (prev[l], prev[u], prev, prev)
        : (prev[u] <= -3) ?
            (0, 0, 0, 0)
        : 
            (
                (simplify_lower(prev, (prev[l] + 3)/6)),
                (simplify_upper(prev, (prev[u] + 3)/6)),
                (replace_lower(prev, (prev[l] + 3)/6)),
                (replace_upper(prev, (prev[u] + 3)/6))
            )
        );
}
