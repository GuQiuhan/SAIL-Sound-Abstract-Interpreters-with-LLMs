transformer deeppoly{
    HardTanh -> 
        ((prev[l] >= 1)) ? (1, 1, 1, 1) :
        ((prev[u] <= -1)) ? (-1, -1, -1, -1) :
        ((prev[l] >= -1) & (prev[u] <= 1)) ? (prev[l], prev[u], prev, prev) :
        ((prev[l] < -1) & (prev[u] > 1)) ? (-1, 1,
            ((prev - prev[l]) * (1 / (prev[u] - prev[l]))) * (1 - (-1)) + (-1),
            ((prev - prev[l]) * (1 / (prev[u] - prev[l]))) * (1 - (-1)) + (-1)
        ) :
        ((prev[l] < -1) & (prev[u] <= 1)) ? (
            -1,
            prev[u],
            ((prev - prev[l]) * ((prev[u] - (-1)) / (prev[u] - prev[l]))) + (-1),
            ((prev - prev[l]) * ((prev[u] - (-1)) / (prev[u] - prev[l]))) + (-1)
        ) :
        ((prev[l] >= -1) & (prev[u] > 1)) ? (
            prev[l],
            1,
            ((prev - prev[l]) * ((1 - prev[l]) / (prev[u] - prev[l]))) + prev[l],
            ((prev - prev[l]) * ((1 - prev[l]) / (prev[u] - prev[l]))) + prev[l]
        ) :
        (undef, undef, undef, undef);
}
