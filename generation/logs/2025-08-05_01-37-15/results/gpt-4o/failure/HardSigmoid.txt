transformer deeppoly{
    HardSigmoid ->
        (prev[l] >= 3)
            ? (1, 1, 1, 1)
        : (prev[u] <= -3)
            ? (0, 0, 0, 0)
        : (
            0.2 * prev[l] + 0.5,
            0.2 * prev[u] + 0.5,
            (prev[u] <= 3) ? (0.2 * prev + 0.5) : (1 + ((prev - 3) * 0)),
            (prev[l] >= -3) ? (0.2 * prev + 0.5) : (0 + ((prev + 3) * 0))
        );
}
