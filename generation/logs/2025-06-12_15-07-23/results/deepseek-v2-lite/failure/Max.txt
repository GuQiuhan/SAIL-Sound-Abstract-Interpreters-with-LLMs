transformer deepz{
    Max -> ((prev_0[l] <= prev_1[l]) ?
                ((prev_0[l]), (prev_0[u]), (prev_0[z])) :
                ((prev_0[u] < prev_1[l]) ?
                    ((prev_1[l]), (prev_1[u]), (prev_1[z])) :
                    ((prev_0[u] >= prev_1[u]) ?
                        ((prev_1[u]), (prev_1[u]), (prev_1[z])) :
                        ((prev_0[u]), (prev_0[u]), (prev_0[z] + prev_1[z]))));
}
