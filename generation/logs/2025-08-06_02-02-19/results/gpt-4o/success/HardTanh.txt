transformer deeppoly{
    HardTanh ->
        (
            (prev[l] >= 1)
                ? (1, 1, 1, 1)
                : (
                    (prev[u] <= -1)
                        ? (-1, -1, -1, -1)
                        : (
                            max(-1, prev[l]),
                            min(1, prev[u]),
                            ((prev - prev[l]) * (min(1, prev[u]) - max(-1, prev[l])) / (prev[u] - prev[l])) + max(-1, prev[l]) - ((prev - prev[l]) * (max(-1, prev[l])) / (prev[u] - prev[l])),
                            ((prev - prev[l]) * (min(1, prev[u]) - max(-1, prev[l])) / (prev[u] - prev[l])) + max(-1, prev[l])
                        )
                )
        );
}
