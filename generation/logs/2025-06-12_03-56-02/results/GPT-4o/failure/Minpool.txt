transformer deeppoly{
    Minpool ->
        (min(prev_0[l], prev_1[l], ..., prev_N[l]),
         min(prev_0[u], prev_1[u], ..., prev_N[u]),
         (prev_0[L].when(prev_0[l] <= prev_1[l] ... && prev_0[l] <= prev_N[l])
          .else(prev_1[L].when(prev_1[l] <= prev_0[l] ... && prev_1[l] <= prev_N[l]))
          ...
          .else(prev_N[L])),
         (prev_0[U].when(prev_0[u] <= prev_1[u] ... && prev_0[u] <= prev_N[u])
          .else(prev_1[U].when(prev_1[u] <= prev_0[u] ... && prev_1[u] <= prev_N[u]))
          ...
          .else(prev_N[U]))
        );
}
