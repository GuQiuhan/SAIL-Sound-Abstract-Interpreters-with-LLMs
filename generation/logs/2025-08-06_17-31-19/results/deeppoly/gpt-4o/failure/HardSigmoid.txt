transformer deeppoly{
    HardSigmoid ->
        ((prev[l]) >= 2) ? (1, 1, 1, 1) :
        ((prev[u]) <= -2) ? (0, 0, 0, 0) :
        (
            max(0, min(1, 0.25 * prev[l] + 0.5)),
            max(0, min(1, 0.25 * prev[u] + 0.5)),
            max(0, min(1, 0.25 * prev[l] + 0.5)),
            max(0, min(1, 0.25 * prev[u] + 0.5))
        );
}
