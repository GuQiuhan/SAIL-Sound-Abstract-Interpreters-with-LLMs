transformer deeppoly{
    HardSwish ->
        (prev[l] >= 3) ?
            (prev[l], prev[u], prev, prev)
        : (prev[u] <= -3) ?
            (0, 0, 0, 0)
        :
            (
                f1(prev[l]),
                f3(prev),
                slope(prev[l], prev[u]) * prev + intercept(prev[l], prev[u]),
                f3(prev)
            );
}
