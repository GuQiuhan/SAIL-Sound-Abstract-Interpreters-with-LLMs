    HardSigmoid -> (
        (prev[l] >= 3) ? (1,
                          1,
                          1,
                          1)
        : ((prev[u] <= -3) ? (0,
                              0,
                              0,
                              0)
        : (f2(prev[l]),
           f1(prev[u]),
           prev * (1/6) + 0.5,
           prev * (1/6) + 0.5))
    );
}
