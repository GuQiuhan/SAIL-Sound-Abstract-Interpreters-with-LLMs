transformer deeppoly{
    HardSigmoid ->
        (prev[l] >= 3) ? (1, 1, 1, 1)
        : (prev[u] <= -3) ? (0, 0, 0, 0)
        : (
            max(0, min(1, 0.2 * prev[l] + 0.5)),
            max(0, min(1, 0.2 * prev[u] + 0.5)),
            (prev[l] >= 3) ? 1 : ((prev[u] <= -3) ? 0 : (0.2 * prev + 0.5)),
            (prev[l] >= 3) ? 1 : ((prev[u] <= -3) ? 0 : (((max(0, min(1, 0.2 * prev[u] + 0.5)) - max(0, min(1, 0.2 * prev[l] + 0.5))) / (prev[u] - prev[l])) * (prev - prev[l]) + max(0, min(1, 0.2 * prev[l] + 0.5))))
        );
}
