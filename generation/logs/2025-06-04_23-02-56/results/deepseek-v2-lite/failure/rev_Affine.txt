    RevAffine -> ((prev[l]) <= 0) ? (0, 0, 0, 0) : ((prev[u]) >= 0) ? (0, 0, 0, 0) : (min(prev[u], 0-prev[l]), max(prev[u], 0-prev[l]), prev*(prev[u]+prev[l])/(prev[u]-prev[l]) - (((2*prev[u])*prev[l])/(prev[u]-prev[l])), prev*(prev[u]+prev[l])/(prev[u]-prev[l]) + (((2*prev[u])*prev[l])/(prev[u]-prev[l])));
}
