transformer deeppoly{
    Neuron_list_mult ->
        // Assume we have prev1, prev2 as the input neurons
        // Calculate the multiplied result and its bounds
        let result = prev1 * prev2;
        let l = min(prev1[l] * prev2[l], prev1[l] * prev2[u], prev1[u] * prev2[l], prev1[u] * prev2[u]);
        let u = max(prev1[l] * prev2[l], prev1[l] * prev2[u], prev1[u] * prev2[l], prev1[u] * prev2[u]);
        let L = prev1[L] * prev2[l] + prev1[l] * prev2[L];
        let U = prev1[U] * prev2[u] + prev1[u] * prev2[U];
        (l, u, L, U);
}
