transformer deeppoly{
    Affine -> (
        // Assuming W is the weight matrix and b is the bias vector
        // W and b are constants for the affine operation
        // curr = prev * W + b
        // To simplify, let's assume W is a single row vector and b is a scalar
        (prev[l] * W[l] + b[l], prev[u] * W[u] + b[u], prev * W[L] + b[L], prev * W[U] + b[U]),
        where W[l], W[u], W[L], W[U] are the bounds of weights,
              b[l], b[u], b[L], b[U] are the bounds of biases
    );
}
