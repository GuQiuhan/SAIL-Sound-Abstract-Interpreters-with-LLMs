transformer deeppoly{
    HardSwish ->
        (prev[l] >= 3) ? (prev[l], prev[u], prev, prev) :
        (prev[u] <= -3) ? (0, 0, 0, 0) :
        (0,
         max(f1(prev[l]), f1(prev[u])),
         0,
         (prev[u] <= 0) ? (slope(prev[l], prev[u]) * prev + intercept(prev[l], prev[u])) : f3(prev)
        );
}
