transformer deeppoly{
    Abs ->
        (prev_0[l] >= 0) ?
            (prev_0[l], prev_0[u], prev_0, prev_0)
        :
        (prev_0[u] <= 0) ?
            (0 - prev_0[u], 0 - prev_0[l], 0 - prev_0, 0 - prev_0)
        :
            (
                0,
                max(prev_0[u], 0 - prev_0[l]),
                prev_0,
                (prev_0[u] / (prev_0[u] - prev_0[l])) * prev_0 - ((prev_0[u] * prev_0[l]) / (prev_0[u] - prev_0[l]))
            );
}
