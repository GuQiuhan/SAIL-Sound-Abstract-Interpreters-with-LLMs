transformer deeppoly{
    HardSwish ->
        ((prev[l]) >= 3) ?
            ((prev[l]), (prev[u]), (prev), (prev))
        : ((prev[u]) <= -3) ?
            (0, 0, 0, 0)
        : (
            // general case
            (min(0, prev[l]), max(prev[u], prev[u] * (prev[u] + 3) / 6),
             prev * (min(1, max(0, (prev + 3) / 6))),
             prev * (max(1, min(0, (prev + 3) / 6))))
        );
}
