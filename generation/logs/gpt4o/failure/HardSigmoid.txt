   HardSigmoid ->
        ((prev[u]) <= -3) ? (0, 0, 0, 0) :
        ((prev[l]) >= 3) ? (1, 1, 1, 1) :
        (((prev[l]) >= -3) & ((prev[u]) <= 3)) ? (
            ((prev[l] + 3.0) / 6.0),
            ((prev[u] + 3.0) / 6.0),
            ((prev + 3.0) / 6.0),
            ((prev + 3.0) / 6.0)
        ) :
        ((prev[l]) < -3) ? (
            0,
            ((prev[u] + 3.0) / 6.0),
            0,
            ((prev + 3.0) / 6.0)
        ) :
        (
            ((prev[l] + 3.0) / 6.0),
            1,
            ((prev + 3.0) / 6.0),
            1
        );
}
