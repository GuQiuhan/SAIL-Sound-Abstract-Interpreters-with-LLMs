transformer deeppoly{
    HardSwish ->
        ((prev[l]) >= 3) ? (
            (prev[l]),
            (prev[u]),
            (prev),
            (prev)
        ) :
        ((prev[u]) <= -3) ? (
            0,
            0,
            0,
            0
        ) :
        (
            // lower bound
            ((prev[l]) >= 0) ?
                (prev[l] * (prev[l] + 3) / 6)
            :
                0,
            // upper bound
            ((prev[u]) <= 0) ?
                (prev[u] * (prev[u] + 3) / 6)
            :
                (prev[u]),
            // lower poly
            ((prev[l]) >= 0) ?
                (prev * (prev + 3) / 6)
            :
                0,
            // upper poly
            ((prev[u]) <= 0) ?
                (prev * (prev + 3) / 6)
            :
                (prev)
        );
}
