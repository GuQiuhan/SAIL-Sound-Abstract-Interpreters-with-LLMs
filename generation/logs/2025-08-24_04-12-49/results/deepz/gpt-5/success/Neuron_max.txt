transformer deepz{
    Neuron_max -> (prev_0[l] >= prev_1[u]) ?
        (prev_0[l], prev_0[u], prev_0[z]) :
        ((prev_1[l] >= prev_0[u]) ?
            (prev_1[l], prev_1[u], prev_1[z]) :
            (max(prev_0[l], prev_1[l]),
             max(prev_0[u], prev_1[u]),
             ((max(prev_0[l], prev_1[l]) + max(prev_0[u], prev_1[u])) / 2) + (((max(prev_0[u], prev_1[u]) - max(prev_0[l], prev_1[l])) / 2) * eps)));
}
