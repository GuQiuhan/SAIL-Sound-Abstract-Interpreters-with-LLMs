transformer deeppoly{
    HardSigmoid ->
        (prev[l] >= 1) ? (1, 1, 1, 1)
        : (prev[u] <= -1) ? (0, 0, 0, 0)
        : ((prev[l] >= -1) and (prev[u] <= 1)) ?
            (((prev[l] + 1) / 2), ((prev[u] + 1) / 2), ((prev + 1) / 2), ((prev + 1) / 2))
        : ((prev[l] < -1) and (prev[u] <= 1)) ?
            (0, ((prev[u] + 1) / 2), 0, ((prev + 1) / 2))
        : ((prev[l] >= -1) and (prev[u] > 1)) ?
            (((prev[l] + 1) / 2), 1, ((prev + 1) / 2), 1)
        : (0, 1, 0, 1);
}
